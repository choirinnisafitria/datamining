def diss(x,y):
  a = 4
  b = 0
  for col in ["pelabuhan",	"kapal_berangkat",	"kapal_tiba",	"penumpang_naik",	"penumpang_turun", "tanggal"	]:
    if p[col][x] == p[col][y]:
      b+=1
    return (a-b)/a

print(diss(1,1))

# Menghitung jarak data Numeric dengan cara Euclidean
def num_p(a,b,c):
    result = 0
    for col in [hasil1, hasil2]:
        result = math.pow(abs(col[a] - col[b]), c)
    return result**(1/c)

print(num_p(1,2,2))
print(num_p(1,3,2))
